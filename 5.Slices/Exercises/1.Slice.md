### Create an empty Slice and append an string to that slice.
============================================================
```
package main

import "fmt"

func main() {
	// Create an empty slice.
	// ... Its length is 0.
	cars := []string{}
	fmt.Println(len(cars))

	// Append a string and the slice now has a length of 1.
	cars = append(cars, "bmw")
	fmt.Println(len(cars))
}
```
